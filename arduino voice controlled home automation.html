so,we have completed the wiring and after the wiring you may upload the program to the arduino the program is given below you may copy paste this program in arduino ide and upload it to arduino

        #define fan 2
        #define bulb 3
        
        void setup() {
          // put your setup code here, to run once:
          Serial.begin(9600);
          pinMode(fan, OUTPUT);
          pinMode(bulb, OUTPUT);
        }
        
        void loop() {
          // put your main code here, to run repeatedly:
          if(Serial.available() == 1)
          {
            String val = Serial.readString();
            Serial.println(val);
            if(val == "bulb to off")
            {
              digitalWrite(fan, HIGH);
            }
            if(val == "bulb to on")
            {
              digitalWrite(fan, LOW);
            }
            if(val == "bulb off")
            {
              digitalWrite(bulb, HIGH);
            }
            if(val == "bulb on")
            {
              digitalWrite(bulb, LOW);
            }
            if(val == "all of")
            {
              digitalWrite(fan, HIGH);
              digitalWrite(bulb, HIGH);
            }
            if(val == "all on")
            {
              digitalWrite(bulb, LOW);
              digitalWrite(fan, LOW);
            }
          }
        }

after you upload the program you are ready to use the voice controlled home automation
